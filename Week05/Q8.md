## 2.	Write a C Program to find LCM of 3 given numbers using recursive functions.

```c
#include <stdio.h>

// Recursive function to calculate GCD
int gcd(int a, int b) {
    if (b == 0)
        return a;
    return gcd(b, a % b);
}

// Function to calculate LCM of two numbers using GCD
int lcm(int a, int b) {
    return (a * b) / gcd(a, b);
}

int main() {
    int num1, num2, num3, result;

    printf("Enter three positive integers: ");
    scanf("%d %d %d", &num1, &num2, &num3);

    // Calculate LCM of three numbers
    result = lcm(num1, num2);
    result = lcm(result, num3);

    printf("LCM of %d, %d, and %d is %d\n", num1, num2, num3, result);

    return 0;
}

```
