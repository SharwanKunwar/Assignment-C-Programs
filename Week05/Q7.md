## 1.	Write a C Program to find the power of a given number using non recursive functions.

```c
#include <stdio.h>

// Function to calculate power using loop (non-recursive)
int power(int base, int exponent) {
    int result = 1;

    for (int i = 1; i <= exponent; i++) {
        result *= base;
    }

    return result;
}

int main() {
    int base, exponent;

    printf("Enter base: ");
    scanf("%d", &base);

    printf("Enter exponent (non-negative): ");
    scanf("%d", &exponent);

    if (exponent < 0) {
        printf("This program supports only non-negative exponents.\n");
    } else {
        int result = power(base, exponent);
        printf("%d raised to the power %d is %d\n", base, exponent, result);
    }

    return 0;
}

```
