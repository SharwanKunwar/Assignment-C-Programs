## 2.	Write a C program to reverse a given integer number using non recursive functions.

```c
#include <stdio.h>

// Function to reverse the number using a loop
int reverseNumber(int n) {
    int reversed = 0;
    while (n != 0) {
        int digit = n % 10;
        reversed = reversed * 10 + digit;
        n /= 10;
    }
    return reversed;
}

int main() {
    int number;

    printf("Enter an integer number: ");
    scanf("%d", &number);

    int original = number;
    if (number < 0)
        number = -number;  // Make positive for reversing

    int reversed = reverseNumber(number);

    if (original < 0)
        reversed = -reversed;  // Keep the sign

    printf("Reversed number: %d\n", reversed);

    return 0;
}

```
