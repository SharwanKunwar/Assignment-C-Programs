## 2.	Write a C Program to find GCD of 3 given numbers using non recursive functions.

```c
#include <stdio.h>

// Function to find GCD of two numbers (non-recursive)
int gcd(int a, int b) {
    int temp;
    while (b != 0) {
        temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

int main() {
    int num1, num2, num3, result;

    printf("Enter three positive integers: ");
    scanf("%d %d %d", &num1, &num2, &num3);

    // First find GCD of num1 and num2, then find GCD of the result with num3
    result = gcd(num1, num2);
    result = gcd(result, num3);

    printf("GCD of %d, %d, and %d is %d\n", num1, num2, num3, result);

    return 0;
}

```
