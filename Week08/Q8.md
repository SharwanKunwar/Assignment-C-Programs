## 2.	Write a C Program to calculate the sum of elements of each row and each column in given matrix using functions.

```c
#include <stdio.h>

// Function to input a matrix
void inputMatrix(int matrix[10][10], int rows, int cols) {
    printf("Enter elements of the %dx%d matrix:\n", rows, cols);
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }
}

// Function to calculate and display row-wise sum
void rowSum(int matrix[10][10], int rows, int cols) {
    printf("\nSum of each row:\n");
    for (int i = 0; i < rows; i++) {
        int sum = 0;
        for (int j = 0; j < cols; j++) {
            sum += matrix[i][j];
        }
        printf("Row %d sum = %d\n", i + 1, sum);
    }
}

// Function to calculate and display column-wise sum
void columnSum(int matrix[10][10], int rows, int cols) {
    printf("\nSum of each column:\n");
    for (int j = 0; j < cols; j++) {
        int sum = 0;
        for (int i = 0; i < rows; i++) {
            sum += matrix[i][j];
        }
        printf("Column %d sum = %d\n", j + 1, sum);
    }
}

int main() {
    int matrix[10][10], rows, cols;

    printf("Enter number of rows and columns: ");
    scanf("%d%d", &rows, &cols);

    inputMatrix(matrix, rows, cols);
    rowSum(matrix, rows, cols);
    columnSum(matrix, rows, cols);

    return 0;
}

```
