## 2.	Write a C program to find sum of main and opposite diagonal elements of a matrix using functions.

```c
#include <stdio.h>

// Function to input matrix
void inputMatrix(int matrix[10][10], int size) {
    printf("Enter elements of the %dx%d matrix:\n", size, size);
    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }
}

// Function to calculate main diagonal sum
int mainDiagonalSum(int matrix[10][10], int size) {
    int sum = 0;
    for (int i = 0; i < size; i++) {
        sum += matrix[i][i];
    }
    return sum;
}

// Function to calculate opposite diagonal sum
int oppositeDiagonalSum(int matrix[10][10], int size) {
    int sum = 0;
    for (int i = 0; i < size; i++) {
        sum += matrix[i][size - i - 1];
    }
    return sum;
}

int main() {
    int matrix[10][10], size;

    printf("Enter the size of the square matrix: ");
    scanf("%d", &size);

    inputMatrix(matrix, size);

    int mainSum = mainDiagonalSum(matrix, size);
    int oppSum = oppositeDiagonalSum(matrix, size);

    printf("Sum of main diagonal elements = %d\n", mainSum);
    printf("Sum of opposite diagonal elements = %d\n", oppSum);

    return 0;
}

```
