## 2.	Write a C Program to find whether given matrix is symmetric or not .

```c
#include <stdio.h>

int main() {
    int matrix[10][10], transpose[10][10];
    int i, j, size;
    int isSymmetric = 1;

    // Input size (square matrix)
    printf("Enter the size of the square matrix (max 10): ");
    scanf("%d", &size);

    // Input matrix elements
    printf("Enter the elements of %dx%d matrix:\n", size, size);
    for (i = 0; i < size; i++) {
        for (j = 0; j < size; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }

    // Create transpose of matrix
    for (i = 0; i < size; i++) {
        for (j = 0; j < size; j++) {
            transpose[j][i] = matrix[i][j];
        }
    }

    // Compare original matrix with its transpose
    for (i = 0; i < size; i++) {
        for (j = 0; j < size; j++) {
            if (matrix[i][j] != transpose[i][j]) {
                isSymmetric = 0;
                break;
            }
        }
        if (!isSymmetric)
            break;
    }

    // Output result
    if (isSymmetric)
        printf("The matrix is symmetric.\n");
    else
        printf("The matrix is not symmetric.\n");

    return 0;
}

```
