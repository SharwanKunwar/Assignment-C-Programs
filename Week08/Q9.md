## 1.	Write a C Program to find largest and smallest numbers in a given list of integers.

```c
#include <stdio.h>

int main() {
    int arr[100], n, i;
    int largest, smallest;

    // Input size
    printf("Enter the number of elements: ");
    scanf("%d", &n);

    // Input elements
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    // Initialize largest and smallest with first element
    largest = smallest = arr[0];

    // Find largest and smallest
    for (i = 1; i < n; i++) {
        if (arr[i] > largest)
            largest = arr[i];
        if (arr[i] < smallest)
            smallest = arr[i];
    }

    // Output results
    printf("Largest number = %d\n", largest);
    printf("Smallest number = %d\n", smallest);

    return 0;
}

```
