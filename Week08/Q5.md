## 1.	Write a C Program to put odd and even elements of given arrays into two separate arrays.

```c
#include <stdio.h>

int main() {
    int arr[100], odd[100], even[100];
    int n, i, oddCount = 0, evenCount = 0;

    // Input size of array
    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);

    // Input elements
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);

        // Separate into odd and even
        if (arr[i] % 2 == 0) {
            even[evenCount++] = arr[i];
        } else {
            odd[oddCount++] = arr[i];
        }
    }

    // Print even array
    printf("\nEven elements:\n");
    for (i = 0; i < evenCount; i++) {
        printf("%d ", even[i]);
    }

    // Print odd array
    printf("\nOdd elements:\n");
    for (i = 0; i < oddCount; i++) {
        printf("%d ", odd[i]);
    }

    printf("\n");
    return 0;
}

```
