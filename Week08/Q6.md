## 2.	Write a C Program to find addition and subtraction of matrices using functions.

```c
#include <stdio.h>

// Function to input matrix
void inputMatrix(int matrix[10][10], int rows, int cols) {
    int i, j;
    printf("Enter elements (%dx%d):\n", rows, cols);
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }
}

// Function to display matrix
void displayMatrix(int matrix[10][10], int rows, int cols) {
    int i, j;
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            printf("%d ", matrix[i][j]);
        }
        printf("\n");
    }
}

// Function to add two matrices
void addMatrix(int a[10][10], int b[10][10], int result[10][10], int rows, int cols) {
    int i, j;
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            result[i][j] = a[i][j] + b[i][j];
        }
    }
}

// Function to subtract two matrices
void subtractMatrix(int a[10][10], int b[10][10], int result[10][10], int rows, int cols) {
    int i, j;
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            result[i][j] = a[i][j] - b[i][j];
        }
    }
}

int main() {
    int a[10][10], b[10][10], sum[10][10], diff[10][10];
    int rows, cols;

    printf("Enter number of rows and columns of matrices: ");
    scanf("%d%d", &rows, &cols);

    printf("Matrix A:\n");
    inputMatrix(a, rows, cols);

    printf("Matrix B:\n");
    inputMatrix(b, rows, cols);

    addMatrix(a, b, sum, rows, cols);
    subtractMatrix(a, b, diff, rows, cols);

    printf("\nSum of matrices:\n");
    displayMatrix(sum, rows, cols);

    printf("\nDifference of matrices (A - B):\n");
    displayMatrix(diff, rows, cols);

    return 0;
}

```
