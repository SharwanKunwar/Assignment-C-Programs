## 1.	Write a C program to print the repeated elements with frequency count in an array.

```c
#include <stdio.h>

int main() {
    int arr[100], freq[100];
    int size, i, j, count;

    printf("Enter the size of the array: ");
    scanf("%d", &size);

    printf("Enter %d elements:\n", size);
    for (i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
        freq[i] = -1; // Initialize frequency array
    }

    // Count frequency of each element
    for (i = 0; i < size; i++) {
        count = 1;
        for (j = i + 1; j < size; j++) {
            if (arr[i] == arr[j]) {
                count++;
                freq[j] = 0; // Mark as counted
            }
        }
        if (freq[i] != 0) {
            freq[i] = count;
        }
    }

    // Print repeated elements and their frequency
    printf("\nRepeated elements with frequency:\n");
    for (i = 0; i < size; i++) {
        if (freq[i] > 1) {
            printf("%d occurs %d times\n", arr[i], freq[i]);
        }
    }

    return 0;
}

```
