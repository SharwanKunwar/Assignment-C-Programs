## 1.	Write a C Program to find the number of non repeated elements in an array.

```c
#include <stdio.h>

int main() {
    int arr[100], freq[100];
    int n, i, j, count, nonRepeatedCount = 0;

    // Input size
    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);

    // Input elements
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
        freq[i] = -1; // Initialize frequency array
    }

    // Count frequency of each element
    for (i = 0; i < n; i++) {
        count = 1;
        for (j = i + 1; j < n; j++) {
            if (arr[i] == arr[j]) {
                count++;
                freq[j] = 0; // Mark as counted
            }
        }
        if (freq[i] != 0) {
            freq[i] = count;
        }
    }

    // Count and print non-repeated elements
    printf("\nNon-repeated elements:\n");
    for (i = 0; i < n; i++) {
        if (freq[i] == 1) {
            printf("%d ", arr[i]);
            nonRepeatedCount++;
        }
    }

    printf("\nTotal number of non-repeated elements: %d\n", nonRepeatedCount);

    return 0;
}

```
