## 1.	Write a C Program to find 2nd largest and 2nd smallest element in an array.

```c
#include <stdio.h>
#include <limits.h>

int main() {
    int arr[100], size;
    int i;
    int largest, second_largest;
    int smallest, second_smallest;

    printf("Enter size of the array: ");
    scanf("%d", &size);

    printf("Enter %d elements:\n", size);
    for (i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }

    if (size < 2) {
        printf("Array must contain at least two elements.\n");
        return 1;
    }

    // Initialize variables
    largest = second_largest = INT_MIN;
    smallest = second_smallest = INT_MAX;

    for (i = 0; i < size; i++) {
        // For largest and second largest
        if (arr[i] > largest) {
            second_largest = largest;
            largest = arr[i];
        } else if (arr[i] > second_largest && arr[i] != largest) {
            second_largest = arr[i];
        }

        // For smallest and second smallest
        if (arr[i] < smallest) {
            second_smallest = smallest;
            smallest = arr[i];
        } else if (arr[i] < second_smallest && arr[i] != smallest) {
            second_smallest = arr[i];
        }
    }

    // Check if second largest/smallest were found
    if (second_largest == INT_MIN)
        printf("No second largest element found.\n");
    else
        printf("Second largest element is: %d\n", second_largest);

    if (second_smallest == INT_MAX)
        printf("No second smallest element found.\n");
    else
        printf("Second smallest element is: %d\n", second_smallest);

    return 0;
}
```
